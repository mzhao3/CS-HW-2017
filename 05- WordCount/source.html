def tally(L):
    d = {}
    for i in range(len(L)):
        if L[i].lower() in d:
             d[L[i].lower()] = d[L[i].lower()] + 1
        else:
             d[L[i].lower()] = 1
    return d

def makeTag(d, tag):
    return "<" + tag + ">" + str(d) + "</" + tag + ">"

def makeRow(row, tag):
    final = "<tr>"
    for i in range(len(row)):
        final = final + makeTag(row[i], tag)
    final = final + "</tr>"
    return final

def toHTMLTable(header, L):
    table =  "<table>" + "\n"
    table = table + makeRow(header, "th") + "\n"
    for row in L:
        table = table + makeRow(row,"td") + "\n"
    table = table + "</table>"
    return table


text = open("waroftheworlds.txt", "r").read()


text= text.replace("--", " ")
ONE = text.split()

for i in range(len(ONE)):
    ONE[i] = ONE[i].strip('[],.!?;:"{}()_')
    ONE[i] = ONE[i].strip("'")
ONE = filter(None, ONE)
unique = tally(ONE)
ONE = sorted(unique.keys())

heading = ["ID", "Word", "count"]

listofthings = []

    listofthings.append([i, ONE[i], unique[ONE[i]]])

usedonce = 0

STAT = str(character) + " characters in the text file. <br>" + str(len(ONE)) + " words in the book. <br>" +str(len(unique)) + " unique words in the book. <br>" + str(over250) + " words that are used over 250 times. <br>" +str(usedonce) + " words that are used once. <br>" + str(over15) + " words that are over 15 letters long. <br>"

TITLE= "Word Count"
