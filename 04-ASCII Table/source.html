def makeTag(d, tag):
    return "<" + tag + ">" + str(d) + ":" + chr(d) + "</" + tag + ">"

def makeRow(row, tag):
    final = "<tr>"
    for i in range(len(row)):
        final = final + makeTag(row [i], tag)
    final = final + "</tr>"
    return final
